SET (PLATFORM_CROSS_COMPILER "arm-linux-")

IF (NOT EXISTS ${PLATFORM_SDK_PATH})
    EXECUTE_PROCESS(
        COMMAND mkdir -p ${PLATFORM_SDK_PATH}
        COMMAND tar zxf ${CMAKE_CURRENT_LIST_DIR}/${PLATFORM_LIBS_GTAR} -C ${PLATFORM_SDK_PATH}
        )
ENDIF(NOT EXISTS ${PLATFORM_SDK_PATH})

IF (NOT EXISTS ${EXTERNAL_LIBS_PATH}) 
    EXECUTE_PROCESS(
        COMMAND ln -s ${PLATFORM_SDK_PATH}/rootfs/usr/lib ${EXTERNAL_LIBS_PATH}
        COMMAND tar zxf ${CMAKE_CURRENT_LIST_DIR}/${EXTERNAL_LIBS_GTAR} -C ${PLATFORM_SDK_PATH}/rootfs
        )
ENDIF(NOT EXISTS ${EXTERNAL_LIBS_PATH})
